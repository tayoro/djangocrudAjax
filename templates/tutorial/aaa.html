$(document).ready(function(){ $('#submit').on('click', function(e){ e.preventDefault(); e.stopPropagation(); let formdata = new FormData(); formdata.append('name', $('[name=name] option:selected').val()); formdata.append('category', $('[name=category] option:selected').val()); formdata.append('created_at', $('[name=created_at] option:selected').val()); formdata.append('updated_at', $('[name=updated_at] option:selected').val()); $('#image').on("change",function(){ var file = this.files[0] if(formdata){ formdata.append("image", file); } }) }) })
